        <input type="text" id="searchName" placeholder="輸入品項關鍵字...">
        <button onclick="fetchData()">搜尋雲端資料</button>

        <div id="resultArea" style="display:none;">
            <div class="item-header">
                <h2 id="displayTitle" style="margin:0;"></h2>
            </div>
            
            <div id="groupsContainer"></div>

            <div class="grand-total">
                <div style="font-size: 14px; opacity: 0.8;">總出貨數量</div>
                <div id="grandSum" style="font-size: 28px; font-weight: bold;"></div>
            </div>
        </div>
    </div>

<script>
    const SHEET_ID = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vSKJ-bbty6XrC42lhDt4IZkx30Xd5suv7asS7ujAncXMnv_AnHBRNuXbxU_HeH3sNxduS22z8I_rqF0/pubhtml'; 
    const SHEET_URL = `https://docs.google.com/spreadsheets/d/${SHEET_ID}/gviz/tq?tqx=out:json`;

    async function fetchData() {
        const target = document.getElementById('searchName').value.trim();
        if(!target) return;

        try {
            const resp = await fetch(SHEET_URL);
            const text = await resp.text();
            const json = JSON.parse(text.substr(47).slice(0, -2));
            // 搜尋 B 欄品項
            const row = json.table.rows.find(r => r.c[1] && r.c[1].v.includes(target));

            if (row) {
                render(row);
            } else {
                alert("找不到該品項");
            }
        } catch (e) {
            alert("讀取失敗，請確認網路與試算表權限");
        }
    }

    function render(row) {
        const name = row.c[1].v;
        const detailStr = row.c[2].v; // 店家 數量, 店家 數量
        const ruleStr = (row.c[4] && row.c[4].v) ? row.c[4].v : ""; // 分組規則欄位 (E欄)
        
        document.getElementById('displayTitle').innerText = name;
        const container = document.getElementById('groupsContainer');
        container.innerHTML = "";

        // 1. 解析店家資料
        const allShops = detailStr.split(',').map(s => {
            const parts = s.trim().split(/\s+/);
            return { name: parts[0], qty: parseInt(parts[1]) || 0, used: false };
        });

        // 2. 解析分組規則 (例如: "中港 / 北冰 / 安星+小賴")
        const groups = ruleStr ? ruleStr.split('/').map(g => g.trim()) : [];

        let totalSum = 0;

        if (groups.length > 0) {
            // --- 有分組模式 ---
            groups.forEach((groupRule, index) => {
                const targets = groupRule.split('+').map(t => t.trim());
                let groupHtml = "";
                let groupSum = 0;

                allShops.forEach(shop => {
                    if (targets.some(t => shop.name.includes(t))) {
                        groupHtml += `<div class="row"><span>${shop.name}</span><span>${shop.qty}</span></div>`;
                        groupSum += shop.qty;
                        shop.used = true;
                    }
                });

                if (groupHtml) {
                    const box = document.createElement('div');
                    box.className = `group-box color-${index % 4}`;
                    box.innerHTML = `<strong>📍 組別：${groupRule}</strong>${groupHtml}<div class="subtotal">小計: ${groupSum}</div>`;
                    container.appendChild(box);
                    totalSum += groupSum;
                }
            });
        }

        // 3. 處理剩餘未分組的店家 (或完全沒設規則時)
        let remainHtml = "";
        let remainSum = 0;
        allShops.forEach(shop => {
            if (!shop.used) {
                remainHtml += `<div class="row"><span>${shop.name}</span><span>${shop.qty}</span></div>`;
                remainSum += shop.qty;
            }
        });

        if (remainHtml) {
            const box = document.createElement('div');
            box.className = "group-box no-group";
            box.innerHTML = `<strong>📍 ${groups.length > 0 ? '其他店家' : '店家清單'}</strong>${remainHtml}<div class="subtotal">小計: ${remainSum}</div>`;
            container.appendChild(box);
            totalSum += remainSum;
        }

        document.getElementById('grandSum').innerText = totalSum;
        document.getElementById('resultArea').style.display = 'block';
    }
</script>
